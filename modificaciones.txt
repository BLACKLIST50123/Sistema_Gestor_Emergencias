--- Archivo: backend/services/authMiddleware.js ---
Ya existía verificarToken() (valida el JWT). Le agregué una segunda
función, requireRole():

    function requireRole(...rolesPermitidos) {
      return (req, res, next) => {
        if (!req.operador || !rolesPermitidos.includes(req.operador.rol)) {
          return res.status(403).json({
            error: `Acción no permitida para el rol '${req.operador ? req.operador.rol : "desconocido"}'`
          });
        }
        next();
      };
    }

¿Cómo funciona? Es una "función que fabrica middlewares". Le pasas
los roles que SÍ pueden entrar, ej: requireRole("administrador"), y
te devuelve una función que Express ejecuta antes que la ruta. Si el
rol del operador logueado (que viene del JWT, ver auth.js) no está
en la lista, corta la petición con 403 (prohibido).

Nota sobre los ESTADOS: tu tabla Recursos en PostgreSQL ya tenía un
CHECK que solo permite ('disponible','ocupado','mantenimiento',

Si tu profesor pide literalmente el texto 'En Emergencia' en la

--- 4.1 Match y Trigger: asignar recurso a una alerta ---

Archivo: backend/routes/alertas.js (ruta NUEVA)

    router.put("/alertas/:id/asignar-recurso", requireRole("operador","administrador"), async (req, res) => {
      const { id_recurso } = req.body;

      // 1) Cambia el estado del recurso en su BD dueña (PostgreSQL)
      const recursoResult = await pgPool.query(
        `UPDATE Recursos SET estado = 'ocupado' WHERE id_recurso = $1 AND activo = TRUE RETURNING *`,
        [id_recurso]
      );
      const recurso = recursoResult.rows[0];

      // Replicidad: reutiliza syncService para propagar el nuevo estado
      const replicas = await sincronizarRecurso(recurso);

      // 2) Vincula el recurso a la alerta y la pasa a 'en_atencion' (Cassandra)
      await cassandraClient.execute(
        `UPDATE Alertas SET id_recurso_asignado = ?, estado = ?, fecha_actualizacion = ? WHERE id_alerta = ?`,
        [id_recurso, "en_atencion", new Date(), id_alerta],
        { prepare: true }
      );

      res.json({ mensaje: "...", recurso, replicas });
    });

Es el "Match" (el operador elige un recurso libre) + "Trigger" (ese
mismo request dispara el cambio de estado) que pediste, todo en un
solo endpoint para que sea fácil de explicar.

--- 4.2 Cierre: liberar el recurso al cerrar la alerta ---

Archivo: backend/routes/alertas.js (ruta que YA existía, modificada)

    router.put("/alertas/:id/estado", requireRole("operador","administrador"), async (req, res) => {
      const { estado } = req.body;
      await cassandraClient.execute(`UPDATE Alertas SET estado = ? ...`, ...);

      if (estado === "cerrada") {
        // busca qué recurso tenía asignado esta alerta...
        const idRecursoAsignado = ...;
        if (idRecursoAsignado) {
          // ...y lo libera de vuelta a 'disponible'
          const recursoResult = await pgPool.query(
            `UPDATE Recursos SET estado = 'disponible' WHERE id_recurso = $1 RETURNING *`,
            [idRecursoAsignado]
          );
          await sincronizarRecurso(recursoResult.rows[0]);
        }
      }

      res.json({ ..., habilitarEvidencias: estado === "cerrada" });
    });
